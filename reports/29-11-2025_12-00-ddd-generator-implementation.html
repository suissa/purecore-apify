🔧 Utilitários de Validação

validateEntity

export function validateEntity<T>( schema: z.ZodSchema<T>, data: unknown, entityName: string ): T { try { return schema.parse(data); } catch (error) { if (error instanceof z.ZodError) { console.error(`❌ Erro de validação em ${entityName}:`); error.errors.forEach(err => { console.error(` ${err.path.join('.')}: ${err.message}`); }); } throw error; } }

validateField

export function validateField<T>( fieldName: string, validator: (value: unknown) => T, value: unknown ): T { try { return validator(value); } catch (error) { console.error(`❌ Erro no campo '${fieldName}':`, error); throw error; } }
